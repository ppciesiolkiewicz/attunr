# Voice Instruction Generator — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CLI tool that generates farinelli voice guidance audio via ElevenLabs TTS and uploads to Vercel Blob.

**Architecture:** SSMLGenerator builds SSML from exercise config params (`maxCount`), CLI calls ElevenLabs API for audio + timestamps, upload command pushes to Vercel Blob. Exercise config gains optional `voice` field with Blob URLs.

**Tech Stack:** TypeScript (tsx), ElevenLabs REST API (raw fetch), @vercel/blob, dotenv

**Spec:** `docs/superpowers/specs/2026-03-21-voice-instruction-generator-design.md`

### Task 5: Create ssml-generator.ts with farinelli SSML generation

The SSMLGenerator imports the journey config and builds SSML from exercise params.

- [ ] **Step 1: Create ssml-generator.ts**

```ts
// scripts/ssml-generator.ts
import { journey } from "../src/constants/journey";

/**
 * Look up a farinelli exercise by chapter slug + exercise slug.
 * Returns the maxCount from the exercise config.
 */
function getFarinelliMaxCount(chapterSlug: string, exerciseSlug: string): number {
  const chapter = journey.getChapterBySlug(chapterSlug);
  if (!chapter) throw new Error(`Chapter "${chapterSlug}" not found`);

  const allExercises = chapter.warmup
    ? [chapter.warmup, ...chapter.stages].flatMap((s) => s.exercises)
    : chapter.stages.flatMap((s) => s.exercises);

  const exercise = allExercises.find((e) => e.slug === exerciseSlug);
  if (!exercise) throw new Error(`Exercise "${exerciseSlug}" not found in chapter "${chapterSlug}"`);
  if (exercise.exerciseTypeId !== "breathwork-farinelli") {
    throw new Error(`Exercise "${exerciseSlug}" is not a farinelli exercise (type: ${exercise.exerciseTypeId})`);
  }

  return exercise.maxCount;
}

/**
 * Generate a counting sequence with breaks: "1 <break/> 2 <break/> ... N"
 */
function countSequence(n: number, breakTime: string = "1s"): string {
  return Array.from({ length: n }, (_, i) => `${i + 1}`)
    .join(` <break time="${breakTime}"/> `);
}

/**
 * Generate SSML for a farinelli breathing exercise.
 *
 * Pattern: countdown (3, 2, 1), then cycles from 4 to maxCount.
 * Each cycle: Inhale (count to cycle#), Hold (count to cycle#), Exhale (always count to 8).
 * Starts at cycle 4 because counts 1-3 are too short for meaningful breathing.
 */
export function generateFarinelliSSML(chapterSlug: string, exerciseSlug: string): string {
  const maxCount = getFarinelliMaxCount(chapterSlug, exerciseSlug);
  const startCycle = 4;

  const lines: string[] = [
    `<speak>`,
    `  3 <break time="1s"/> 2 <break time="1s"/> 1`,
    `  <break time="1s"/>`,
  ];

  for (let cycle = startCycle; cycle <= maxCount; cycle++) {
    const isLast = cycle === maxCount;

    lines.push(``);
    lines.push(`  Inhale. <break time="1s"/> ${countSequence(cycle)}`);
    lines.push(`  <break time="1.5s"/>`);
    lines.push(`  Hold. <break time="1s"/> ${countSequence(cycle)}`);
    lines.push(`  <break time="1.5s"/>`);
    lines.push(`  Exhale. <break time="1s"/> ${countSequence(8)}`);

    if (!isLast) {
      lines.push(`  <break time="2s"/>`);
    }
  }

  lines.push(`</speak>`);
  return lines.join("\n");
}

/**
 * Generate SSML for a given exercise. Dispatches to the appropriate generator
 * based on exercise type.
 */
export function generateSSML(chapterSlug: string, exerciseSlug: string): string {
  const chapter = journey.getChapterBySlug(chapterSlug);
  if (!chapter) throw new Error(`Chapter "${chapterSlug}" not found`);

  const allExercises = chapter.warmup
    ? [chapter.warmup, ...chapter.stages].flatMap((s) => s.exercises)
    : chapter.stages.flatMap((s) => s.exercises);

  const exercise = allExercises.find((e) => e.slug === exerciseSlug);
  if (!exercise) throw new Error(`Exercise "${exerciseSlug}" not found in chapter "${chapterSlug}"`);

  switch (exercise.exerciseTypeId) {
    case "breathwork-farinelli":
      return generateFarinelliSSML(chapterSlug, exerciseSlug);
    default:
      throw new Error(`No SSML generator for exercise type "${exercise.exerciseTypeId}"`);
  }
}
```

### Task 6: Create voice-instruction-generator.ts CLI

**Files:**
- Create: `scripts/voice-instruction-generator.ts`

- [ ] **Step 1: Create the CLI with generate and upload subcommands**

```ts
// scripts/voice-instruction-generator.ts
import dotenv from "dotenv";
import { resolve } from "path";
dotenv.config({ path: resolve(__dirname, "../.env.local") });

// Register tsconfig paths so transitive @/ imports resolve correctly
import "tsconfig-paths/register";

import { existsSync, mkdirSync, writeFileSync, readFileSync } from "fs";
import path from "path";
import { put } from "@vercel/blob";
import { generateSSML } from "./ssml-generator";
import { voiceSettings } from "./voice-instruction-settings";

const VOICE_OUTPUT_DIR = path.join(__dirname, "voice-output");

// ── ElevenLabs API ──────────────────────────────────────────────────────────

interface ElevenLabsTimestampResponse {
  audio_base64: string;
  alignment: {
    characters: string[];
    character_start_times_seconds: number[];
    character_end_times_seconds: number[];
  };
}

async function generateAudio(
  ssml: string,
  voiceId: string,
  modelId: string,
): Promise<{ audioBuffer: Buffer; timestamps: ElevenLabsTimestampResponse["alignment"] }> {
  const apiKey = process.env.ELEVENLABS_API_KEY;
  if (!apiKey) throw new Error("ELEVENLABS_API_KEY not set in .env.local");

  const url = `https://api.elevenlabs.io/v1/text-to-speech/${voiceId}/with-timestamps`;

  const response = await fetch(url, {
    method: "POST",
    headers: {
      "xi-api-key": apiKey,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      text: ssml,
      model_id: modelId,
      enable_ssml_parsing: true,
    }),
  });

  if (!response.ok) {
    const errorText = await response.text();
    throw new Error(`ElevenLabs API error (${response.status}): ${errorText}`);
  }

  const data: ElevenLabsTimestampResponse = await response.json();
  const audioBuffer = Buffer.from(data.audio_base64, "base64");

  return { audioBuffer, timestamps: data.alignment };
}

// ── Generate subcommand ─────────────────────────────────────────────────────

async function generate(chapterSlug: string, exerciseSlug: string) {
  console.log(`Generating voice for ${chapterSlug}/${exerciseSlug}...`);

  // 1. Generate SSML from exercise config
  const ssml = generateSSML(chapterSlug, exerciseSlug);
  console.log("SSML generated:");
  console.log(ssml);

  // 2. Call ElevenLabs
  console.log(`\nCalling ElevenLabs (voice: ${voiceSettings.instructionVoiceId})...`);
  const { audioBuffer, timestamps } = await generateAudio(
    ssml,
    voiceSettings.instructionVoiceId,
    voiceSettings.modelId,
  );

  // 3. Save files
  const outputDir = path.join(VOICE_OUTPUT_DIR, chapterSlug, exerciseSlug);
  mkdirSync(outputDir, { recursive: true });

  const mp3Path = path.join(outputDir, "instruction.mp3");
  const timestampsPath = path.join(outputDir, "instruction-timestamps.json");

  writeFileSync(mp3Path, audioBuffer);
  writeFileSync(timestampsPath, JSON.stringify(timestamps, null, 2));

  console.log(`\nSaved:`);
  console.log(`  ${mp3Path}`);
  console.log(`  ${timestampsPath}`);
  console.log(`  Audio size: ${(audioBuffer.length / 1024).toFixed(1)} KB`);
}

// ── Upload subcommand ───────────────────────────────────────────────────────

async function upload(chapterSlug: string, exerciseSlug: string) {
  const token = process.env.ATTUNR_BLOB_READ_WRITE_TOKEN;
  if (!token) throw new Error("ATTUNR_BLOB_READ_WRITE_TOKEN not set in .env.local");

  const outputDir = path.join(VOICE_OUTPUT_DIR, chapterSlug, exerciseSlug);

  const mp3Path = path.join(outputDir, "instruction.mp3");
  const timestampsPath = path.join(outputDir, "instruction-timestamps.json");

  if (!existsSync(mp3Path) || !existsSync(timestampsPath)) {
    throw new Error(`Files not found in ${outputDir}. Run "generate" first.`);
  }

  console.log(`Uploading ${chapterSlug}/${exerciseSlug} to Vercel Blob...`);

  const blobPrefix = `voice/${chapterSlug}/${exerciseSlug}`;

  const [mp3Blob, timestampsBlob] = await Promise.all([
    put(`${blobPrefix}/instruction.mp3`, readFileSync(mp3Path), {
      access: "public",
      token,
      contentType: "audio/mpeg",
    }),
    put(`${blobPrefix}/instruction-timestamps.json`, readFileSync(timestampsPath), {
      access: "public",
      token,
      contentType: "application/json",
    }),
  ]);

  console.log(`\nUploaded! Add these URLs to the exercise config:\n`);
  console.log(`voice: {`);
  console.log(`  instructionUrl: "${mp3Blob.url}",`);
  console.log(`  instructionTimestampsUrl: "${timestampsBlob.url}",`);
  console.log(`}`);
}

// ── CLI entry point ─────────────────────────────────────────────────────────

async function main() {
  const [subcommand, chapterSlug, exerciseSlug] = process.argv.slice(2);

  if (!subcommand || !chapterSlug || !exerciseSlug) {
    console.error("Usage: npx tsx scripts/voice-instruction-generator.ts <generate|upload> <chapterSlug> <exerciseSlug>");
    process.exit(1);
  }

  switch (subcommand) {
    case "generate":
      await generate(chapterSlug, exerciseSlug);
      break;
    case "upload":
      await upload(chapterSlug, exerciseSlug);
      break;
    default:
      console.error(`Unknown subcommand: ${subcommand}. Use "generate" or "upload".`);
      process.exit(1);
  }
}

main().catch((err) => {
  console.error(err.message);
  process.exit(1);
});
```
